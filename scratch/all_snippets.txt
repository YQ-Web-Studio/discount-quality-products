add_action('rest_api_init', function () {

        'methods'             => 'POST',

        'permission_callback' => '__return_true',

        'methods'             => array('GET', 'POST'),
        'callback'            => 'handle_wishlist_request',

/**
 * Handle public user registration for the headless storefront.
 *
 * Expected payload:
 * - name
 * - email
 * - password
 */
function handle_user_registration_request($request) {
    $params = $request->get_json_params();

    $name = isset($params['name']) ? sanitize_text_field($params['name']) : '';
    $email = isset($params['email']) ? sanitize_email($params['email']) : '';
    $password = isset($params['password']) ? (string) $params['password'] : '';

    if (!$name || !$email || !$password) {
        return new WP_Error('invalid_registration_payload', 'Name, email and password are required.', array('status' => 400));
    }

    if (!is_email($email)) {
        return new WP_Error('invalid_email', 'Please enter a valid email address.', array('status' => 400));
    }

    if (email_exists($email)) {
        return new WP_Error('email_exists', 'An account already exists for that email address.', array('status' => 409));
    }

    $username_base = sanitize_user(current(explode('@', $email)), true);
    if (!$username_base) {
        $username_base = 'customer';
    }

    $username = $username_base;
    $suffix = 1;

    while (username_exists($username)) {
        $username = $username_base . $suffix;
        $suffix++;
    }

    $user_id = wp_insert_user(array(
        'user_login'   => $username,
        'user_pass'    => $password,
        'user_email'   => $email,
        'display_name' => $name,
        'nickname'     => $name,
        'role'         => 'subscriber',
    ));

    if (is_wp_error($user_id)) {
        return new WP_Error('registration_failed', $user_id->get_error_message(), array('status' => 500));
    }

    return rest_ensure_response(array(
        'success' => true,
        'user_id' => $user_id,
        'message' => 'Account created successfully.',
    ));
}

/**
 * Handle GET and POST requests for the wishlist.
 * 
 * @param WP_REST_Request $request
 * @return WP_REST_Response|WP_Error
 */
function handle_wishlist_request($request) {
    $user_id = get_current_user_id();
    
    // Safety check (redundant due to permission_callback but good practice)
    if (!$user_id) {
        return new WP_Error('not_logged_in', 'User must be logged in to manage wishlist.', array('status' => 401));
    }

    // Retrieve current wishlist (array of product IDs)
    $wishlist = get_user_meta($user_id, 'wishlist_products', true);
    
    // Ensure it's an array (handles first-time setup or empty meta)
    if (!is_array($wishlist)) {
        $wishlist = array();
    }

    // Handle POST request (Add or Remove)
    if ($request->get_method() === 'POST') {
        $params = $request->get_json_params();
        $product_id = isset($params['product_id']) ? (int) $params['product_id'] : (int) $request->get_param('product_id');

        if (!$product_id) {
            return new WP_Error('invalid_product_id', 'A valid product_id is required.', array('status' => 400));
        }

        if (in_array($product_id, $wishlist)) {
            // Remove the product ID
            $wishlist = array_values(array_diff($wishlist, array($product_id)));
            $action = 'removed';
        } else {
            // Add the product ID
            $wishlist[] = $product_id;
            $action = 'added';
        }

        // Save back to user_meta
        update_user_meta($user_id, 'wishlist_products', $wishlist);
        
        return rest_ensure_response(array(
            'success' => true,
            'action'  => $action,
            'data'    => $wishlist
        ));
    }

    // Handle GET request (Retrieve)
    return rest_ensure_response(array(
        'success' => true,
        'data'    => $wishlist
    ));
}

    register_rest_route('custom/v1', '/delete-account', array(
        'methods'  => 'POST',
        'callback' => 'dq_delete_current_account',
        'permission_callback' => function () {
            return is_user_logged_in();
        },
        'args' => array(
            'current_password' => array(
                'required' => true,
                'type'     => 'string',
            ),
        ),
    ));

function dq_delete_current_account(WP_REST_Request $request) {
    $user = wp_get_current_user();

    if (!$user || empty($user->ID)) {
        return new WP_REST_Response(array(
            'error' => 'Please sign in again.',
        ), 401);
    }

    $current_password = (string) $request->get_param('current_password');

    if ($current_password === '') {
        return new WP_REST_Response(array(
            'error' => 'Please enter your current password.',
        ), 400);
    }

    if (!wp_check_password($current_password, $user->user_pass, $user->ID)) {
        return new WP_REST_Response(array(
            'error' => 'Your current password does not match our records.',
        ), 403);
    }

    if (!function_exists('wp_delete_user')) {
        require_once ABSPATH . 'wp-admin/includes/user.php';
    }

    $deleted = wp_delete_user($user->ID);

    if (!$deleted) {
        return new WP_REST_Response(array(
            'error' => 'We could not delete your account right now.',
        ), 500);
    }

    return new WP_REST_Response(array(
        'success' => true,
        'message' => 'Your account has been deleted.',
    ), 200);
}

        'args' => array(

        ),

function custom_get_payment_tokens() {
    $user_id = get_current_user_id();
    $tokens = WC_Payment_Tokens::get_customer_tokens($user_id);
    $formatted_tokens = array();

    foreach ($tokens as $token) {
        if ($token->get_type() === 'CC') {
            $formatted_tokens[] = array(
                'id' => $token->get_id(),
                'brand' => $token->get_card_type(),
                'last4' => $token->get_last4(),
                'expiry' => $token->get_expiry_month() . '/' . $token->get_expiry_year()
            );
        }
    }
    return rest_ensure_response($formatted_tokens);
}

function custom_delete_payment_token($request) {
    $user_id = get_current_user_id();
    $token_id = $request['id'];
    $token = WC_Payment_Tokens::get($token_id);

    if (!$token || $token->get_user_id() !== $user_id) {
        return new WP_Error('not_found', 'Payment token not found.', array('status' => 404));
    }

    if (WC_Payment_Tokens::delete($token_id)) {
        return rest_ensure_response(array('success' => true));
    }
    return new WP_Error('delete_failed', 'Could not delete token.', array('status' => 500));
}

function custom_create_setup_intent($request) {
    $user_id = get_current_user_id();
    
    if (!class_exists('WC_Stripe_API')) {
        return new WP_Error('stripe_missing', 'WooCommerce Stripe plugin is not active.', array('status' => 500));
    }

    // Try to get existing Stripe Customer ID
    $customer_id = get_user_meta($user_id, '_stripe_customer_id', true);
    $user = get_userdata($user_id);

    try {
        if (!$customer_id) {
            // Create a new Stripe Customer
            $customer = WC_Stripe_API::request(array(
                'email' => $user->user_email,
                'name'  => $user->first_name . ' ' . $user->last_name,
            ), 'customers');
            
            if (!empty($customer->id)) {
                $customer_id = $customer->id;
                update_user_meta($user_id, '_stripe_customer_id', $customer_id);
            } else {
                return new WP_Error('stripe_error', 'Failed to create customer.', array('status' => 500));
            }
        }

        // Create Setup Intent
        $setup_intent = WC_Stripe_API::request(array(
            'customer' => $customer_id,
            'payment_method_types' => array('card'),
        ), 'setup_intents');

        return rest_ensure_response(array(
            'client_secret' => $setup_intent->client_secret
        ));

    } catch (Exception $e) {
        return new WP_Error('stripe_error', $e->getMessage(), array('status' => 500));
    }
}

function custom_add_payment_method($request) {
    $user_id = get_current_user_id();
    $payment_method_id = $request->get_param('payment_method_id');

    if (empty($payment_method_id)) {
        return new WP_Error('missing_param', 'Payment method ID is required.', array('status' => 400));
    }

    if (!class_exists('WC_Stripe_API')) {
        return new WP_Error('stripe_missing', 'WooCommerce Stripe plugin is not active.', array('status' => 500));
    }

    try {
        // Retrieve the Payment Method from Stripe
        $payment_method = WC_Stripe_API::request(array(), "payment_methods/{$payment_method_id}", 'GET');
        
        if (empty($payment_method->id) || $payment_method->type !== 'card') {
            return new WP_Error('invalid_method', 'Invalid payment method.', array('status' => 400));
        }

        // Try to get existing Stripe Customer ID
        $customer_id = get_user_meta($user_id, '_stripe_customer_id', true);
        
        // If no customer ID exists, create one
        if (!$customer_id) {
            $user = get_userdata($user_id);
            $customer = WC_Stripe_API::request(array(
                'email' => $user->user_email,
                'name'  => $user->first_name . ' ' . $user->last_name,
            ), 'customers', 'POST');
            
            if (!empty($customer->id)) {
                $customer_id = $customer->id;
                update_user_meta($user_id, '_stripe_customer_id', $customer_id);
            } else {
                return new WP_Error('stripe_error', 'Failed to create Stripe customer.', array('status' => 500));
            }
        }

        // ATTACH THE PAYMENT METHOD TO THE CUSTOMER
        $attach_response = WC_Stripe_API::request(array(
            'customer' => $customer_id
        ), "payment_methods/{$payment_method_id}/attach", 'POST');

        if (isset($attach_response->error)) {
            return new WP_Error('stripe_attach_error', $attach_response->error->message, array('status' => 500));
        }

        // Create WooCommerce Payment Token
        $token = new WC_Payment_Token_CC();
        $token->set_token($payment_method->id);
        $token->set_gateway_id('stripe');
        $token->set_card_type(strtolower($payment_method->card->brand));
        $token->set_last4($payment_method->card->last4);
        $token->set_expiry_month($payment_method->card->exp_month);
        $token->set_expiry_year($payment_method->card->exp_year);
        $token->set_user_id($user_id);
        
        // Save the token
        if (!$token->save()) {
            return new WP_Error('db_error', 'Failed to save token to database.', array('status' => 500));
        }

        return rest_ensure_response(array(
            'success' => true,
            'token_id' => $token->get_id()
        ));

    } catch (Exception $e) {
        return new WP_Error('stripe_error', $e->getMessage(), array('status' => 500));
    }
}

/**
 * Title: WooCommerce REST API Global Attribute Filter Interceptor
 * Description: Intercepts Next.js API product requests via pre_get_posts to allow multi-select taxonomy filtering.
 */

add_action( 'pre_get_posts', function( $query ) {
    // Only target the main query on WooCommerce REST API products endpoint
    if ( ! is_admin() && strpos( $_SERVER['REQUEST_URI'], '/wp-json/wc/v3/products' ) !== false ) {
        
        $tax_query = $query->get( 'tax_query' );
        if ( ! is_array( $tax_query ) ) {
            $tax_query = array();
        }
        
        $has_attribute_filter = false;
        
        // Loop through $_GET parameters directly
        foreach ( $_GET as $key => $value ) {
            if ( strpos( $key, 'pa_' ) === 0 && ! empty( $value ) ) {
                $terms = array_map( 'trim', explode( ',', $value ) );
                
                $tax_query[] = array(
                    'taxonomy' => $key,
                    'field'    => 'name',
                    'terms'    => $terms,
                    'operator' => 'IN',
                );
                
                $has_attribute_filter = true;
            }
        }
        
        if ( $has_attribute_filter ) {
            if ( count( $tax_query ) > 1 ) {
                $tax_query['relation'] = 'AND';
            }
            $query->set( 'tax_query', $tax_query );
        }
    }
} );

/**
 * Snippet Title: Safe Auto-Delete Product Images (Hash & Variant Aware)
 * Description: Deletes attachment files from disk/Cloudflare R2 only if no other products share the same asset hash.
 */

add_action('before_delete_post', 'dqp_safe_delete_product_images', 10, 1);

function dqp_safe_delete_product_images($post_id) {
    // Only execute if the post type being deleted is a WooCommerce product
    if (get_post_type($post_id) !== 'product') {
        return;
    }

    $image_ids = array();

    // 1. Fetch the primary Featured Image ID
    $featured_id = get_post_thumbnail_id($post_id);
    if ($featured_id) {
        $image_ids[] = intval($featured_id);
    }

    // 2. Fetch the Product Gallery image IDs
    $gallery = get_post_meta($post_id, '_product_image_gallery', true);
    if (!empty($gallery)) {
        $gallery_ids = explode(',', $gallery);
        $image_ids = array_merge($image_ids, array_filter(array_map('intval', $gallery_ids)));
    }

    // Deduplicate the array in case an image is listed in both featured and gallery
    $image_ids = array_unique($image_ids);

    // 3. Loop through each image and verify relationships before deleting
    foreach ($image_ids as $attachment_id) {
        if ($attachment_id <= 0) {
            continue;
        }

        // If the image hash is shared by any other product, skip deletion
        if (dqp_is_image_shared_by_others($attachment_id, $post_id)) {
            continue; 
        }

        // Safe to delete permanently—bypasses the trash bin
        // Media Cloud will intercept this hook and instantly delete the file from Cloudflare R2
        wp_delete_attachment($attachment_id, true);
    }
}

/**
 * Helper function to query the database for shared image references
 */
function dqp_is_image_shared_by_others($attachment_id, $current_product_id) {
    global $wpdb;

    // Check A: Is this attachment ID used as a featured image (_thumbnail_id) on any OTHER product?
    $featured_count = $wpdb->get_var($wpdb->prepare(
        "SELECT COUNT(*) FROM {$wpdb->postmeta} WHERE meta_key = '_thumbnail_id' AND meta_value = %d AND post_id != %d",
        $attachment_id,
        $current_product_id
    ));

    if (intval($featured_count) > 0) {
        return true;
    }

    // Check B: Is this attachment ID listed inside the gallery string (_product_image_gallery) of any OTHER product?
    $gallery_count = $wpdb->get_var($wpdb->prepare(
        "SELECT COUNT(*) FROM {$wpdb->postmeta} WHERE meta_key = '_product_image_gallery' AND meta_value LIKE %s AND post_id != %d",
        '%' . $wpdb->esc_like($attachment_id) . '%',
        $current_product_id
    ));

    if (intval($gallery_count) > 0) {
        return true;
    }

    return false; // No dependencies found; the asset is safe to clear out completely
}

            if (isset($option['feedrules']['output_type'])) {

            // Wait, brand is type "pattern" and default "Co" in the option!
            // Let's look at the mapping: brand attribute index is 13:
            // "mattributes" index 13: "brand", type index 13: "pattern", default index 13: "Co".

                $option['feedrules']['attributes'][13] = 'pa_brand';

        'methods' => 'POST',
        'callback' => function() {
            $option_name = 'wf_feed_dqpfeed';
            $option = get_option($option_name);
            if ( ! $option ) {
                return new WP_REST_Response( array( 'success' => false, 'error' => 'Feed option not found in database' ), 404 );
            }

            // Unserialize if it is serialized (WP get_option usually does this automatically)
            $option = maybe_unserialize($option);
            $before = $option;

            // 1. Clean output_type array in feedrules
            if ( isset( $option['feedrules']['output_type'] ) && is_array( $option['feedrules']['output_type'] ) ) {
                $count = count( $option['feedrules']['output_type'] );
                for ( $i = 0; $i < $count; $i++ ) {
                    $option['feedrules']['output_type'][$i] = array(); // Set to empty array
                }
            }

            // 2. Fix Brand mapping (Index 13)
            // Verify first that index 13 matches brand
            if ( isset( $option['feedrules']['mattributes'][13] ) && $option['feedrules']['mattributes'][13] === 'brand' ) {
                $option['feedrules']['type'][13] = 'attribute';
                $option['feedrules']['attributes'][13] = 'pa_brand';
                $option['feedrules']['default'][13] = '';
            }

            // 3. Update option in database
            $updated = update_option( $option_name, $option );

            // 4. Regenerate feed
            $generation_result = 'Not triggered';
            $feed_url = '';
            if ( function_exists( 'woo_feed_generate_feed' ) ) {
                // Trigger feed generation synchronously
                ob_start();
                $feed_url = woo_feed_generate_feed( $option, 'dqpfeed' );
                $generation_result = ob_get_clean();
            }

            return new WP_REST_Response( array(
                'success' => true,
                'db_updated' => $updated,
                'feed_url' => $feed_url,
                'generation_logs' => $generation_result,
                'option_after' => $option
            ), 200 );

        'permission_callback' => '__return_true'
    ) );
